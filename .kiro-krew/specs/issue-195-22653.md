# Issue #195: Implement ACP-based Planning Tab for Kiro CLI v3.0 Compatibility

**Closes #195**

## Executive Summary

This specification outlines the migration from subprocess-based planning to ACP (Agent Communication Protocol) based planning to resolve TUI rendering conflicts and prepare for Kiro CLI v3.0 compatibility. The solution involves integrating the official `acp-go-sdk` package and creating a new Planning tab type with dedicated chat interface, while maintaining backward compatibility through a `plan classic` command.

## Problem Analysis

### Current State Analysis

The existing implementation in `internal/session/planner.go` uses:
- `exec.CommandContext` to spawn `kiro-cli chat --classic --agent planner`
- Subprocess pipe communication via `cmd.StdinPipe()`, `cmd.StdoutPipe()`, `cmd.StderrPipe()`
- Manual process lifecycle management (suspend/resume via SIGSTOP/SIGCONT)
- TUI takeover using `tea.ExecProcess` which causes rendering conflicts

### Root Cause Issues

1. **TUI Conflicts**: `tea.ExecProcess` completely takes over the terminal, disrupting Kiro Krew's interface
2. **Process Management Complexity**: Manual signal handling for suspend/resume introduces race conditions
3. **Kiro CLI v3.0 Incompatibility**: Classic mode (`--classic`) will be deprecated
4. **Error Recovery**: Subprocess crashes require complex recovery mechanisms

### Proposed ACP Solution Benefits

1. **Clean Integration**: ACP connections maintain separate communication channels without TUI conflicts
2. **Native Protocol**: Direct protocol communication eliminates subprocess overhead
3. **Future Compatibility**: ACP is the supported v3.0 communication method
4. **Better Error Handling**: SDK provides built-in retry and error recovery

## Architecture Design

**Proposed (ACP-based)**:

```go
package tui

import (
    "context"
    "fmt"
    "strings"
    "time"
    
    "charm.land/bubbles/v2/textinput"
    "charm.land/bubbles/v2/viewport"
    tea "charm.land/bubbletea/v2"
    "charm.land/lipgloss/v2"
    
    "github.com/coder/acp-go-sdk"
)

const TabTypePlanning TabType = 2

type PlanningTab struct {
    id          string
    acpClient   *acp.Client
    acpSession  *acp.Session
    messages    []ChatMessage
    viewport    viewport.Model
    input       textinput.Model
    width       int
    height      int
    state       PlanningState
    contextInfo *ContextInfo
    styles      *Styles
    ctx         context.Context
    cancel      context.CancelFunc
}

func NewPlanningTab(styles *Styles) (*PlanningTab, error) {
    ctx, cancel := context.WithCancel(context.Background())
    
    // Initialize ACP client
    client, err := acp.NewClient(acp.Config{
        Context: ctx,
    })
    if err != nil {
        cancel()
        return nil, fmt.Errorf("failed to create ACP client: %w", err)
    }
    
    // Start chat session
    session, err := client.StartChat(acp.ChatConfig{
        Agent: "planner",
    })
    if err != nil {
        cancel()
        return nil, fmt.Errorf("failed to start chat session: %w", err)
    }
    
    // Initialize UI components
    input := textinput.New()
    input.Focus()
    input.Prompt = "[planner] > "
    
    viewport := viewport.New(viewport.WithWidth(80), viewport.WithHeight(24))
    
    tab := &PlanningTab{
        id:         generatePlanningTabID(),
        acpClient:  client,
        acpSession: session,
        messages:   make([]ChatMessage, 0),
        viewport:   viewport,
        input:     input,
        state:     StateActive,
        contextInfo: &ContextInfo{},
        styles:    styles,
        ctx:       ctx,
        cancel:    cancel,
    }
    
    // Start message streaming
    go tab.streamMessages()
    
    return tab, nil
}

func (pt *PlanningTab) streamMessages() {
    for response := range pt.acpSession.Stream() {
        switch response.Type {
        case acp.ResponseTypeText:
            pt.appendMessage("assistant", response.Content, false)
        case acp.ResponseTypeJSON:
            if response.Complete {
                pt.appendMessage("assistant", response.Content, true)
            }
        case acp.ResponseTypeContext:
            pt.updateContextInfo(response.Context)
        case acp.ResponseTypeError:
            pt.appendMessage("system", fmt.Sprintf("Error: %v", response.Error), false)
        }
    }
}
```

```go
const maxPlanningTabs = 10

func (tm *TabManager) AddPlanningTab(styles *Styles) error {
    // Enforce 10-tab limit
    planningTabCount := tm.countPlanningTabs()
    if planningTabCount >= maxPlanningTabs {
        return fmt.Errorf("maximum %d Planning tabs allowed", maxPlanningTabs)
    }
    
    planningTab, err := NewPlanningTab(styles)
    if err != nil {
        return fmt.Errorf("failed to create planning tab: %w", err)
    }
    
    tm.AddTab(planningTab)
    tm.SetActiveTab(len(tm.tabs) - 1) // Focus new tab
    return nil
}

func (tm *TabManager) countPlanningTabs() int {
    count := 0
    for _, tab := range tm.tabs {
        if tab.Type() == TabTypePlanning {
            count++
        }
    }
    return count
}
```

#### 3.1 Footer Structure Design

**File**: `internal/tui/footer.go` (NEW)

```go
package tui

import (
    "fmt"
    "os"
    "strings"
    
    "charm.land/lipgloss/v2"
)

type FooterRenderer struct {
    styles *Styles
    width  int
}

func NewFooterRenderer(styles *Styles) *FooterRenderer {
    return &FooterRenderer{
        styles: styles,
    }
}

func (fr *FooterRenderer) RenderTwoRowFooter(inputValue string, contextInfo *ContextInfo, theme string) string {
    // Row 1: Input row (same as current)
    inputRow := fr.renderInputRow(inputValue)
    
    // Row 2: Status row with conditional context info
    statusRow := fr.renderStatusRow(contextInfo, theme)
    
    return inputRow + "\n" + statusRow
}

func (fr *FooterRenderer) renderInputRow(inputValue string) string {
    prompt := fr.styles.Prompt.Render("kiro-krew> ")
    input := inputValue + fr.styles.Prompt.Render("█") // Cursor
    
    return prompt + input
}

func (fr *FooterRenderer) renderStatusRow(contextInfo *ContextInfo, theme string) string {
    var parts []string
    
    // Theme (always present)
    parts = append(parts, fmt.Sprintf("theme: %s", theme))
    
    // Context info (only for planning tabs)
    if contextInfo != nil {
        if contextInfo.ContextUsage != "" {
            parts = append(parts, fmt.Sprintf("ctx: %s", contextInfo.ContextUsage))
        }
        if contextInfo.Model != "" {
            parts = append(parts, fmt.Sprintf("model: %s", contextInfo.Model))
        }
        if contextInfo.Directory != "" {
            parts = append(parts, fmt.Sprintf("📁 %s", contextInfo.Directory))
        }
    }
    
    status := strings.Join(parts, " │ ")
    
    // Apply status row styling
    return fr.styles.StatusRow.Render(status)
}

func (fr *FooterRenderer) Resize(width int) {
    fr.width = width
}
```

```go
    responses []acp.Response
    messages  []string

func (m *MockACPSession) SendMessage(content string) error {

    return nil
}

func (m *MockACPSession) Stream() <-chan acp.Response {
    ch := make(chan acp.Response, len(m.responses))
```
